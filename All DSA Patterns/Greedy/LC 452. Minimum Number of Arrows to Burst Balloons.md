```cpp
  int findMinArrowShots(vector<vector<int>>& points) {
        sort(points.begin(), points.end(), [](const vector<int> &a, 
            const vector<int> &b)
            {
                return a[1] < b[1];
            });

        int j = 0, ans = 1;
        for(int i=1; i<points.size(); i++){
            if(points[j][1] < points[i][0]) {
            ans++;
            j = i;
            }
        }
        return ans;
    }
```
