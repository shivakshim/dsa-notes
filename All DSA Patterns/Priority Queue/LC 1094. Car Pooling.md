```cpp
 bool carPooling(vector<vector<int>>& trips, int capacity) {
        //1. sort by start
        sort(trips.begin(), trips.end(), [](const vector<int> &a, const vector<int>& b){
            return a[1] < b[1];
        });

        //2. use a min heap to track which trip finishes earliest
        int people = 0;
        priority_queue<pair<int, int>, vector<pair<int,int>>, 
        greater<pair<int,int>>> min_heap;
        for(auto t : trips){
            int curr = t[0], start = t[1], end = t[2];

            while(!min_heap.empty() && start >= min_heap.top().first){
                people -= min_heap.top().second;
                min_heap.pop();
            }

            min_heap.push({end, curr});
            people += curr;
            if(people > capacity) return false;
        }
        return true;
    }
```
