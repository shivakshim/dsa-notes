🔗[Problem link](https://leetcode.com/problems/third-maximum-number/description/)
Find Kth greatest DISTINCT element.

---
#### ⌛Brute Force 

- Sort -> loop -> ignore duplicates ->  return n-kth.

#### **Approach 1️⃣ : Using set**
#### 🗝️Key Concept

- Use a set to first filter out any duplicates and then push distinct elements in the array then do normal quick select.

**Steps :
1. Convert array to `set` → vector of unique elements.
2. Compute index for k-th largest: `n - k`.
3. Run standard Quick Select on unique array.

**Time Complexity:**
- `O(n)` average (Quick Select) + `O(n log n)` for set conversion if using `std::set`
- Total: roughly `O(n log n)` with set

**Space Complexity:**
- `O(n)` for set/vector

```cpp
vector<int> getUnique(vector<int>& nums) {
    set<int> s(nums.begin(), nums.end());  // removes duplicates automatically
    return vector<int>(s.begin(), s.end());
}

int partition(vector<int>& arr, int left, int right) {
    int pivot = arr[right];
    int i = left;
    for(int j=left;j<right;j++) {
        if(arr[j] <= pivot) {  // smaller to left
            swap(arr[i], arr[j]);
            i++;
        }
    }
    swap(arr[i], arr[right]);
    return i;
}

int quickSelect(vector<int>& arr, int left, int right, int k) {
    if(left <= right) {
        int pivotIndex = partition(arr, left, right);

        if(pivotIndex == k) return arr[pivotIndex];      // found the k-th smallest
        else if(pivotIndex < k) return quickSelect(arr, pivotIndex + 1, right, k);
        else return quickSelect(arr, left, pivotIndex - 1, k);
    }
    return -1;  // just a safeguard
}

vector<int> uniqueArr = getUnique(nums);
int n = uniqueArr.size();
int kthLargest = quickSelect(uniqueArr, 0, n-1, n-k); 
```

#### **Approach 1️2️⃣ : In Place filtering - no extra set/ds required**

**Steps:**
1. Pick pivot (random).
2. Partition array (< pivot left, > pivot right).
3. Count **distinct elements in right**, ignoring duplicates of pivot.
4. If k falls in right → recurse right.
5. Else if pivot itself is k-th → return pivot.
6. Else adjust k and recurse left.

**Time Complexity:**
- `O(n)` average, `O(n^2)` worst case (random pivot mitigates this)

**Space Complexity:**
- `O(1)` extra (in-place) + recursion stack

**Notes:**
- Tricky, easy to mess up counting duplicates.
- Low ROI in interviews unless memory-restricted.

```cpp
// Partition array in-place
int partition(vector<int>& arr, int left, int right) {
    int pivot = arr[right];
    int i = left;
    for (int j = left; j < right; j++) {
        if (arr[j] < pivot) { // smaller elements to left
            swap(arr[i], arr[j]);
            i++;
        }
    }
    swap(arr[i], arr[right]);
    return i;
}

// Count distinct elements in a range [left, right]
int countDistinct(vector<int>& arr, int left, int right) {
    sort(arr.begin() + left, arr.begin() + right + 1); // to simplify distinct counting
    int count = 0;
    int last = INT_MIN;
    for (int i = left; i <= right; i++) {
        if (arr[i] != last) {
            count++;
            last = arr[i];
        }
    }
    return count;
}

// Fancy Quick Select for k-th largest distinct
int quickSelectDistinct(vector<int>& arr, int left, int right, int k) {
    if (left > right) return -1; // not found

    int pivotIndex = partition(arr, left, right);
    int distinctRight = countDistinct(arr, pivotIndex + 1, right);
    int pivotCount = 1; // pivot itself counts as 1 distinct

    if (distinctRight + pivotCount == k) return arr[pivotIndex]; // pivot is k-th largest distinct
    else if (distinctRight >= k) 
        return quickSelectDistinct(arr, pivotIndex + 1, right, k); // search right
    else 
        return quickSelectDistinct(arr, left, pivotIndex - 1, k - distinctRight - pivotCount); // search left
}
```
