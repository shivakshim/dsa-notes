###### Solution is divided in 2 steps - 

1.Creating a Prefix sum matrix :
```cpp
 int n = mat.size();
        int m = mat[0].size();
        vector<vector<int>> presum;
        presum.resize(n+1, vector<int>(m+1,0));
        //creating 2D prefix sum
        for(int i=0; i<n; i++){
            for(int j=0; j<m; j++){
                presum[i+1][j+1] = mat[i][j] +
                                   presum[i][j+1] +
                                   presum[i+1][j] -
                                   presum[i][j];
            }
        }
```

2. Finding rectangle sum for every answer[i] : (row1 = i-k, col1 = j-k) -> (row2= i+k, col2 = j+k)
  ```cpp
  vector<vector<int>> ans;
        ans.resize(n, vector<int>(m, 0));
        for(int i=0 ; i<n; i++){
            for(int j=0 ; j<m; j++){
               int r1 = max(0, i - k);
               int r2 = min(n - 1, i + k);
               int c1 = max(0, j - k);
               int c2 = min(m - 1, j + k);
               ans[i][j] = presum[r2+1][c2+1] -
                           presum[r1][c2+1] -  //top
                           presum[r2+1][c1] +  //left
                           presum[r1][c1];
            }
        }
      return ans;
    }
